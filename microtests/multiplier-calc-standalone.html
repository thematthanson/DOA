function calculateMultiplier(elapsedMinutes) {
    const milestones = [
        { time: 0, multiplier: 1.0 },
        { time: 12, multiplier: 1.5 },
        { time: 24, multiplier: 2.0 },
        { time: 36, multiplier: 2.5 },
        { time: 48, multiplier: 3.0 },
        { time: 60, multiplier: 3.5 }
    ];
    
    for (let i = milestones.length - 1; i >= 0; i--) {
        if (elapsedMinutes >= milestones[i].time) {
            return milestones[i].multiplier;
        }
    }
    return 1.0;
}
